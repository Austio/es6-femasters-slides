Variables Hoisting

Trap Variables Inside Functions Only

<script type='text/javascript'>
  var a = 1;
  function run(){
    if(true){
      var a = 2;
    }
    console.log(a);        //result is 2
    console.log(this.a);   //result is 1
  }
  run();
  console.log(a); //result is 1
</script>

Function Hoisting

  var bar = undefined;
  function foo(){  //Function declaration hoisted
    //code here
  }

  //SOME CODE GOES HERE

  bar = function(){ //function name hoisted, but variable assignment
                    //doesn't happen until the code gets here
    //code here
  };
</script>
